//PhoneLanguagePreferenceController.java

@Override
public boolean isAvailable() {
    return mContext.getResources().getBoolean(R.bool.config_show_phone_language)
            && mContext.getAssets().getLocales().length > 1;
}

//AbstractPreferenceController.java
/**
* Displays preference in this controller.
/
public void displayPreference(PreferenceScreen screen) {
final String prefKey = getPreferenceKey();
if (TextUtils.isEmpty(prefKey)) {
Log.w(TAG, “Skipping displayPreference because key is empty:” + getClass().getName());
return;
}
if (isAvailable()) {
setVisible(screen, prefKey, true / visible /);
if (this instanceof Preference.OnPreferenceChangeListener) {
final Preference preference = screen.findPreference(prefKey);
preference.setOnPreferenceChangeListener(
(Preference.OnPreferenceChangeListener) this);
}
} else {
setVisible(screen, prefKey, false / visible */);
}
}
